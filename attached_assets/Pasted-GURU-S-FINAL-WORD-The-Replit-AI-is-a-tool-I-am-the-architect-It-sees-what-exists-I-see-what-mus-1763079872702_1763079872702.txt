+  let gains = 0, losses = 0;

// POLYGON LIVE + HISTORICAL
app.get('/api/polygon/:sym', async (req, res) => {
  const { sym } = req.params;
  if (isMarketOpen()) {
    // LIVE WS
    const ws = new WebSocket(`wss://socket.polygon.io/stocks?apiKey=${process.env.POLYGON_KEY}`);
    ws.on('open', () => ws.send(JSON.stringify({ action: 'subscribe', params: `T.${sym}` })));
    ws.on('message', (d) => {
      const msg = JSON.parse(d);
      if (msg.ev === 'T') res.json({ price: msg.p, volume: msg.s, time: getCST() });
    });
  } else {
    // HISTORICAL
    const resp = await axios.get(`https://api.polygon.io/v2/aggs/ticker/${sym}/range/1/day/prev?apiKey=${process.env.POLYGON_KEY}`);
    res.json(resp.data.results[0]);
  }
});
