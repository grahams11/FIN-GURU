tsconst trades = [];

export const logTrade = (symbol, entry, exit = null, status = 'open') => {
  trades.push({ symbol, entry, exit, status, timestamp: Date.now() });
};

export const closeTrade = (symbol, exitPrice) => {
  const trade = trades.find(t => t.symbol === symbol && t.status === 'open');
  if (trade) {
    trade.exit = exitPrice;
    trade.status = 'closed';
    trade.profit = ((exitPrice - trade.entry) / trade.entry) * 100;
  }
};

export const getMetrics = () => {
  const closed = trades.filter(t => t.status === 'closed');
  const wins = closed.filter(t => t.profit > 0);
  const losses = closed.filter(t => t.profit <= 0);
  const winRate = closed.length ? (wins.length / closed.length) * 100 : 0;
  const avgROI = closed.reduce((a, t) => a + t.profit, 0) / closed.length || 0;
  const profitFactor = losses.length ? wins.reduce((a, t) => a + t.profit, 0) / Math.abs(losses.reduce((a, t) => a + t.profit, 0)) : Infinity;
  return { winRate: winRate.toFixed(1), avgROI: avgROI.toFixed(1), profitFactor: profitFactor.toFixed(2), total: closed.length };
};
