let rh: any;

export const initRobinhood = async () => {
  rh = new Robinhood({
    username: process.env.ROBINHOOD_USERNAME,
    password: process.env.ROBINHOOD_PASSWORD
  });
  await rh.login();
  console.log('Robinhood Connected');
};

export const getRobinhoodPositions = async () => {
  const positions = await rh.get_open_positions();
  const options = await rh.get_open_option_positions();
  return [...positions, ...options].map(p => ({
    broker: 'Robinhood',
    symbol: p.symbol || p.underlying_symbol,
    quantity: p.quantity,
    avgCost: p.average_buy_price,
    currentPrice: p.market_value / p.quantity,
    pnl: p.market_value - (p.average_buy_price * p.quantity)
  }));
};
