  const controller = new AbortController()
  let mounted = true

  const scan = async () => {
    // Cancel previous requests
    controller.abort()
    
    // Parallel fetch top 50 tickers only (not full market)
    const symbols = sp500.slice(0, 50)  // or your elite list
    const requests = symbols.map(sym => 
      fetch(`/api/elite-scan?ticker=${sym}`, { signal: controller.signal })
    )

    try {
      const results = await Promise.allSettled(requests)
      const valid = results
        .filter(r => r.status === 'fulfilled')
        .map(r => r.value.json())
      
      if (mounted) setNewTrades(valid)
    } catch {} 
  }

  scan()
  const id = setInterval(scan, 8000)  // refresh every 8s

  return () => {
    mounted = false
    clearInterval(id)
    controller.abort()
  }
