ts// eliteScanner.ts
const getOvernightData = async (sym) => {
  const now = new Date();
  const eod = getEODCache(sym);  // From 4:00 PM
  const overnight = await polygon.getAggs(sym, '1', 'minute', 
    `${now.toISOString().split('T')[0]} 16:00:00`, 
    `${now.toISOString().split('T')[0]} 20:00:00`
  );

  return {
    eodHigh: eod.high,
    eodVolume: eod.volume,
    overnightHigh: Math.max(...overnight.map(b => b.h)),
    overnightVolume: overnight.reduce((s, b) => s + b.v, 0),
    chain: await polygon.getOptionsSnapshot(sym)
  };
};
